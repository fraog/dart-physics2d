void rotateBy(double rads)

Source 

void rotateBy(double rads) {
  double s = sin(rads);
  double c = cos(rads);
  double nx = this.x * c - this.y * s;
  double ny = this.x * s + this.y * c;
  this.x = nx;
  this.y = ny;
}
